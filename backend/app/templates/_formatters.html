{# Reusable Jinja macros for admin PDF/HTML templates #} {%- macro fmt_quota(mb, unlimited=false) -%} {%- if unlimited -%} Unlimited {%- else -%} {%- set v = (mb or 0) | float -%} {%- set sign = '-' if v < 0 else '' -%} {%- set a = (v * -1) if v < 0 else v -%} {# Input is in MB. Output is dynamic KB/MB/GB with base 1024: - < 1 MB => KB (rounded) - 1..1023 MB => MB (rounded) - >= 1024 MB => GB (selalu 2 desimal, mis. 2.00 GB / 1.50 GB) #} {%- if a < (1 / 1024) -%} {{- sign -}}0 KB {%- elif a < 1 -%} {%- set kb = a * 1024 -%} {{- sign -}}{{ kb | round(0, 'common') | int }} KB {%- elif a >= 1024 -%} {%- set gb_str = "%.2f" % (a / 1024) -%} {{- sign -}}{{ gb_str }} GB {%- else -%} {{- sign -}}{{ a | round(0, 'common') | int }} MB {%- endif -%} {%- endif -%} {%- endmacro -%} {%- macro fmt_rp(value_rp) -%} {%- set v = (value_rp if value_rp is not none else 0) | int -%} {{- ('{:,.0f}'.format(v)) | replace(',', '.') -}} {%- endmacro -%}
